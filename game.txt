Upgrades
    // 11 basic upgrades
    *TEMPLATE
		on click:anim glow
    *prestige
        name:Prestige to a new life
        desc:Love Hoi Ki more in a new life. Start fresh.
        cost:(100 + (pow(10, 6) * pow((prestigePoint), 1/0.14)) * 0.75) love
        icon:icons[1,1]
        [i:upgradeClickDisplay %upgradeID="prestige"]
        req:(100 + (pow(10, 6) * pow((prestigePoint), 1/0.14)) * 0.75) <= (love:earned)
        passive:lose :Buildings
        passive:lose :Upgrades
        passive:lose love
        passive:do ascension
        on ascension:
                    if (prestigePoint > 0)
                        multiply yield of :Buildings by (1/((prestigePoint) * (prestigeWeight) * 0.1))
                    end
                    yield pow((love)/pow(10,6),0.14) prestigePoint
                    multiply yield of :Buildings by ((prestigePoint) * (prestigeWeight) * 0.1)
                    toast You Are Prestiged. Now Your Love Gains Is Multiplied By ((prestigePoint) * (prestigeWeight) * 0.1).
                end
